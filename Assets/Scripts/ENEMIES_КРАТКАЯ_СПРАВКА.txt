🎮 МЕХАНИКА:

✅ Преследуют игрока
✅ Лазают по лестницам
✅ Убивают при касании
✅ Застревают в ямах (3 сек)
✅ Возрождаются на спавне


⚙️ КЛЮЧЕВЫЕ ПАРАМЕТРЫ:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

EnemyController:
  Walk Speed ............. 3    (скорость ходьбы)
  Climb Speed ............ 2    (скорость лазания)
  Detection Range ........ 15   (дальность обнаружения)
  Stuck In Hole Time ..... 3    (время в яме, сек)

EnemySpawner:
  Enemy Prefab ........... Префаб врага
  Spawn Points ........... Точки появления
  Max Enemies ............ 5    (максимум врагов)


🎨 GIZMOS:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🔴 Красная сфера ... Зона обнаружения (Detection Range)
🟢 Зеленый круг .... GroundCheck (на земле)
🟡 Желтая линия .... Путь к игроку (при преследовании)
📍 Spawn Points .... Точки спавна врагов


🐛 ПРОБЛЕМЫ:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

❌ Не двигается
   → Walk Speed > 0
   → Игрок в радиусе Detection Range

❌ Не преследует
   → Увеличь Detection Range

❌ Не убивает
   → Оба должны иметь Collider (не Trigger)


💡 БАЛАНСИРОВКА:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Легко:   Walk=2, Range=10, 1-2 врага
Средне:  Walk=3, Range=15, 2-3 врага
Сложно:  Walk=4, Range=20, 4-5 врагов


🔗 ИНТЕГРАЦИЯ С ЯМАМИ:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Добавьте в DiggableBlock.cs:

private void OnTriggerEnter2D(Collider2D collision)
{
    EnemyController enemy = collision.GetComponent<EnemyController>();
    if (enemy != null && isDug && !enemy.IsStuckInHole())
    {
        enemy.StuckInHole();
    }
}
